*** EH Native Fan Component ***

#include "esphome/components/fan/fan.h"

namespace esphome {
namespace currentflow {

class VescFan : public fan::Fan, public Component {
 public:
  // Define what your fan can do (speed, direction, etc.)
  fan::FanTraits get_traits() override {
    auto traits = fan::FanTraits();
    traits.set_speed(true);
    traits.set_direction(true);
    return traits;
  }

  void control(const fan::FanCall &call) override {
    if (call.get_state().has_value()) {
      this->state = *call.get_state();
    }
    if (call.get_speed().has_value()) {
      this->speed = *call.get_speed(); // 0 to 100 percentage
    }
    if (call.get_direction().has_value()) {
      this->direction = *call.get_direction();
    }

    this->write_state_to_vesc();
    this->publish_state();
  }

  void write_state_to_vesc() {
    if (!this->state || this->speed == 0) {
      // Send COMM_SET_DUTY 0 or COMM_SET_CURRENT 0 to VESC
      return;
    }

    // Example: Map 0-100% speed to VESC RPM or Duty Cycle
    float vesc_val = this->speed / 100.0f;
    if (this->direction == fan::FanDirection::REVERSE) {
      vesc_val *= -1;
    }

    // Call your existing UART command function here
    // this->parent_->set_duty_cycle(vesc_val);
  }
};

} // namespace currentflow
} // namespace esphome
