*** Toggle BLE Bridge

switch:
  - platform: template
    name: "VESC Tool Bluetooth Access"
    optimistic: true
    on_turn_on:
      # start_ble() should set enabled_ = true flag and startAdvertising() 
      # _enabled is checked in loop() and onConnect()
      lambda: 'id(my_vesc_bridge).start_ble();' 
    on_turn_off:
      # stop_ble() should set enabled_ = false, disconnect(), 
      # stopAdvertising()
      lambda: 'id(my_vesc_bridge).stop_ble();'

void stop_ble() {
    this->enabled_ = false;
    
    if (this->ble_connected_) {
        // Force the phone to drop the link
        this->ble_server_->disconnect(this->conn_id_);
    }
    
    // Stop broadcasting so the device "disappears"
    this->ble_server_->getAdvertising()->stop();
    
    // Reset state for next time
    this->ble_connected_ = false;
    this->set_mtu(23); 
}

*** Buffered TX ***

static const size_t MAX_BUFFER = 512;
uint8_t buffer[MAX_BUFFER];
size_t index = 0;
uint32_t last_byte_time = 0;

void loop() override {
    // 1. Determine current safe chunk size (MTU - 3 bytes header)
    size_t current_mtu = pServer->getPeerMTU(pServer->getConnId()) - 3;
    // Fallback if not connected or negotiated yet
    if (current_mtu > MAX_BUFFER || current_mtu < 20) current_mtu = 20;

    // 2. Collect bytes
    while (this->available() && index < MAX_BUFFER) {
        buffer[index++] = this->read();
        last_byte_time = millis();

        // Flush immediately if we hit the current MTU limit
        if (index >= current_mtu) {
            flush_buffer(index);
        }
    }

    // 3. Timeout: Flush partial packets after 15ms of silence
    if (index > 0 && (millis() - last_byte_time > 15)) {
        flush_buffer(index);
    }
}

void flush_buffer(size_t length) {
    if (pServer->getConnectedCount() <= 0) return;
    pCharacteristic->setValue(buffer, length);
    pCharacteristic->notify();
    index = 0;
}
